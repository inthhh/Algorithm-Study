vector<int> solution(int n) {
    vector<int> answer;
    for(int i=2; i<=n; ++i){
        if(n%i==0){
            n /= i;
            answer.push_back(i);
            i=1;
        }
        if(n==1) break;
    }
    sort(answer.begin(), answer.end());
    set<int> s(answer.begin(), answer.end());
    vector<int> v(s.begin(), s.end());
    return v;
}
