int solution(vector<int> array) {
    int answer = 1;
    int cnt[101] = {0,};
    for(int i=0; i<101; ++i) { cnt[i] = 1;}
    int max = 1;
    int j=0;
    
    sort(array.begin(), array.end());
    if(array.size()==1) return array[0];
    for(int i=1; i<array.size(); ++i){
        if(array[i-1] == array[i]) {
            cnt[j]++;
        }
        else {
            j++;
        }
        if(max <= cnt[j]) {
            answer = array[i];
            max = cnt[j];
        }
    }
    sort(cnt, cnt+j+1);
    if(cnt[j-1] == cnt[j]) return -1;
    return answer;
}
