vector<int> solution(vector<int> emergency) {
    vector<int> answer;
    int seq[10] = {0,};
    for(int i=0; i<emergency.size(); ++i){
        int max_index = max_element(emergency.begin(), emergency.end()) - emergency.begin();
        seq[max_index] = i+1;
        emergency[max_index] = 0;
    }
    for(int i : seq){
        if(i!=0) answer.push_back(i);
    }
    return answer;
}
