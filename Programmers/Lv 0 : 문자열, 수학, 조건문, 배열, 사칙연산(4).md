vector<string> solution(vector<string> quiz) {
    vector<string> answer;
    for(int i; i<quiz.size(); ++i){
        stringstream ss(quiz[i]);
        int first;
        ss>>first;
        int n; char c;
        while(ss>>c>>n){
            if(c=='+') first += n;
            if(c=='-') first -= n;
            if(c=='=') {
                if(first==n) answer.push_back("O");
                else answer.push_back("X");
            }
        }
    }
    return answer;
}
