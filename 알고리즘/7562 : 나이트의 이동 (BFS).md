#include <iostream>
#include <queue>
using namespace std;

int answer = 0;
int dx[8] = {2, 1, 2, 1, -2, -1, -2, -1};
int dy[8] = {1, 2, -1, -2, 1, 2, -1, -2};
int l;

bool canGo(int visited[301][301], int x, int y){
    if(x<0 || y<0 || x>=l || y>=l) return false;
    if(visited[x][y]==1) return false;
    return true;
}

void bfs(int x, int y, int lx, int ly){
    int visited[301][301]={0};
    int mat[301][301] = {0};
    queue<pair<int, int>> q;
    q.push({x, y});
    while(!q.empty()){
        pair<int, int> a = q.front();
        q.pop();
        for (int i = 0; i < 8; ++i){
            int nx = a.first + dx[i];
            int ny = a.second + dy[i];
            if(nx==x && ny==y) continue;
            if (canGo(visited, nx, ny)){
                visited[nx][ny] = 1;
                q.push({nx, ny});
                mat[nx][ny] = mat[a.first][a.second] + 1;
            }
            if(nx==lx && ny==ly) break;
        }
    }
    answer = mat[lx][ly];
}

int main(){
    int t;
    int x;
    int y;
    int lx;
    int ly;
    cin >> t;
    for (int i = 0; i < t; ++i){
        cin >> l >> x>>y >> lx>>ly;
        bfs(x,y,lx,ly);
        cout << answer << "\n";
        answer = 0;
    }
}```
