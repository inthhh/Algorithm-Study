#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

bool compare(int i, int j){
    return j < i;
}

int main(){
    ios_base::sync_with_stdio(false); cin.tie(NULL);
    vector<int> v[2];
    int n;
    int k;
    int result = 0;
    cin >> n;
    for (int i = 0; i < 2; ++i){
        for (int j = 0; j < n; ++j){
            cin >> k;
            v[i].push_back(k);
        }
    }
    sort(v[0].begin(), v[0].end());
    sort(v[1].begin(), v[1].end(), compare);
    for (int i = 0; i < n; ++i){
        result += v[0][i] * v[1][i];
    }
    cout << result;
}
